files$

FBtoC enhancement
FBtoC's files$ statement supports FSRefs and CFURLRefs. 

#if def _FBtoC	// FBtoC only
dim as FSRef fileRef
fileName = files$( _FSRefOpen,,, fileRef )
#endif

#if def _FBtoC
dim as FSRef parentFolderRef
fileName = files$( _FSRefSave,,, parentFolderRef )
#endif

#if def _FBtoC
dim as CFURLRef fileUrl
fileName = files$( _CFURLRefOpen,,, fileUrl )
#endif

#if def _FBtoC
dim as CFURLRef fileUrl
fileName = files$( _CFURLRefSave,,, fileUrl )
#endif
